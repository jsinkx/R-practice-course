# –æ–¥–Ω–∞ –∏–∑ –≤—ã—à–µ–ø–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—ã—Ö –∫–æ–º–±–∏–Ω–∞—Ü–∏–π.

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–¥—Å—á–µ—Ç–∞ —Ñ–∞–∫—Ç–æ—Ä–∏–∞–ª–∞
factorial <- function(x) {
  if (x == 0) return(1)
  return(prod(1:x))
}

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–¥—Å—á–µ—Ç–∞ —á–∏—Å–ª–∞ —Å–æ—á–µ—Ç–∞–Ω–∏–π C(n, k)
choose <- function(n, k) {
  factorial(n) / (factorial(k) * factorial(n - k))
}

# –û–±—â–µ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–æ–º–±–∏–Ω–∞—Ü–∏–π
total_combinations <- choose(52, 5)

# –ü–∞—Ä–∞ (–æ–¥–Ω–∞ –ø–∞—Ä–∞ –∏ —Ç—Ä–∏ –æ—Å—Ç–∞–ª—å–Ω—ã–µ –∫–∞—Ä—Ç—ã —Ä–∞–∑–Ω—ã–µ –ø–æ –Ω–æ–º–∏–Ω–∞–ª—É)
pairs <- choose(13, 1) * choose(4, 2) * choose(12, 3) * (choose(4, 1)^3)
prob_pair <- pairs / total_combinations

# –î–≤–µ –ø–∞—Ä—ã
two_pairs <- choose(13, 2) * (choose(4, 2)^2) * choose(11, 1) * choose(4, 1)
prob_two_pairs <- two_pairs / total_combinations

# –¢—Ä–æ–π–∫–∞
three_of_a_kind <- choose(13, 1) * choose(4, 3) * choose(12, 2) * (choose(4, 1)^2)
prob_three_of_a_kind <- three_of_a_kind / total_combinations

# –°—Ç—Ä–∏—Ç
straights     <- 10 * (choose(4, 1)^5) - 10 * 4  # –≤—ã—á–∏—Ç–∞–µ–º —Å—Ç—Ä–∏—Ç-—Ñ–ª—ç—à–∏
prob_straight <- straights / total_combinations

# –§–ª—ç—à
flushes    <- choose(4, 1) * choose(13, 5) - 10  # –≤—ã—á–∏—Ç–∞–µ–º —Å—Ç—Ä–∏—Ç-—Ñ–ª—ç—à–∏
prob_flush <- flushes / total_combinations

# –§—É–ª–ª-—Ö–∞—É—Å (—Ç—Ä–∏+–¥–≤–∞)
full_house      <- choose(13, 1) * choose(4, 3) * choose(12, 1) * choose(4, 2)
prob_full_house <- full_house / total_combinations

# –ö–∞—Ä–µ
four_of_a_kind      <- choose(13, 1) * choose(4, 4) * choose(12, 1) * choose(4, 1)
prob_four_of_a_kind <- four_of_a_kind / total_combinations

# –°—Ç—Ä–∏—Ç-—Ñ–ª—ç—à
straight_flushes    <- 10 * choose(4, 1)
prob_straight_flush <- straight_flushes / total_combinations

# –†–æ–π–∞–ª-—Ñ–ª—ç—à
royal_flushes    <- 1 * choose(4, 1)
prob_royal_flush <- royal_flushes / total_combinations

prob_none <- 1 - (prob_pair + prob_two_pairs + prob_three_of_a_kind + 
                    prob_straight + prob_flush + prob_full_house + 
                    prob_four_of_a_kind + prob_straight_flush + prob_royal_flush)

cat("–ü–∞—Ä–∞:", prob_pair, "\n")
## –ü–∞—Ä–∞: 0.422569
cat("–î–≤–µ –ø–∞—Ä—ã:", prob_two_pairs, "\n")
## –î–≤–µ –ø–∞—Ä—ã: 0.04753902
cat("–¢—Ä–æ–π–∫–∞:", prob_three_of_a_kind, "\n")
## –¢—Ä–æ–π–∫–∞: 0.02112845
cat("–°—Ç—Ä–∏—Ç:", prob_straight, "\n")
## –°—Ç—Ä–∏—Ç: 0.003924647
cat("–§–ª—ç—à:", prob_flush, "\n")
## –§–ª—ç—à: 0.001976945
cat("–§—É–ª–ª-—Ö–∞—É—Å:", prob_full_house, "\n")
## –§—É–ª–ª-—Ö–∞—É—Å: 0.001440576
cat("–ö–∞—Ä–µ:", prob_four_of_a_kind, "\n")
## –ö–∞—Ä–µ: 0.000240096
cat("–°—Ç—Ä–∏—Ç-—Ñ–ª—ç—à:", prob_straight_flush, "\n")
## –°—Ç—Ä–∏—Ç-—Ñ–ª—ç—à: 1.539077e-05
cat("–†–æ–π–∞–ª-—Ñ–ª—ç—à:", prob_royal_flush, "\n")
## –†–æ–π–∞–ª-—Ñ–ª—ç—à: 1.539077e-06
cat("–í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Ç–æ–≥–æ, —á—Ç–æ –Ω–µ –≤—ã–ø–∞–¥–µ—Ç –Ω–∏ –æ–¥–Ω–∞ –∫–æ–º–±–∏–Ω–∞—Ü–∏—è:", prob_none, "\n")
## –í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Ç–æ–≥–æ, —á—Ç–æ –Ω–µ –≤—ã–ø–∞–¥–µ—Ç –Ω–∏ –æ–¥–Ω–∞ –∫–æ–º–±–∏–Ω–∞—Ü–∏—è: 0.5011643
